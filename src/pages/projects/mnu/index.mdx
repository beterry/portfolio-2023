export const pageQuery = graphql`
    query mnuQuery{
        mainImage: allFile(filter: {relativeDirectory: {eq: "mnu/images/main"}}) {
            edges {
                node {
                    childImageSharp {
                        fluid {
                            ...GatsbyImageSharpFluid
                        }
                    }
                }
            }
        }
        images: allFile(filter: {relativeDirectory: {eq: "mnu/images"}}, sort: {order: ASC, fields: name}) {
            edges {
                node {
                    childImageSharp {
                        fluid {
                            ...GatsbyImageSharpFluid
                        }
                    }
                }
            }
        }
    }
`
